The newest spec is: a) if either x or y is null, do identical(x, y) b) otherwise call operator ==

@override
bool operator ==(Object other) {
  if (other is Concept?) {
    Concept? concept = other as Concept?;
    if (identical(this, concept!)) {
      return true;
    } else {
      if ((concept as dynamic) == null) {
        return false;
      } else {
        return equals(concept);
      }
    }
  } else {
    return false;
  }
}
